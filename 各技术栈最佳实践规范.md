## 一、Git 规范

### 1. Commit 规范

```
<type>(<scope>): <subject>

feat: 新增用户登录功能
fix(auth): 修复token过期问题
docs: 更新API文档
style: 格式化代码
refactor: 重构用户模块
test: 添加单元测试
chore: 更新依赖
```

### 2. 分支命名

```
feature/user-login
fix/token-expiry
hotfix/critical-bug
release/v1.2.0
```

### 3. Git Flow

```
main/master     → 生产环境
develop         → 开发分支
feature/*       → 功能分支
release/*       → 发布分支
hotfix/*        → 紧急修复
```

## 二、HTML 规范

### 4. 语义化标签

```html
<!-- ✅ 好 -->
<header><nav><main><article><section><footer>

<!-- ❌ 差 -->
<div class="header"><div class="nav">
```

### 5. 属性顺序

```html
<input type="text" name="username" id="username" placeholder="用户名" />
<!-- type → name → id → class → data- → aria- → 其他 -->
```

### 6. 嵌套规则

```html
<!-- 内联元素不能包含块级元素 -->
<!-- ❌ 差 -->
<span><div>错误</div></span>

<!-- ✅ 好 -->
<div><span>正确</span></div>
```

### 7. 资源路径

```html
<!-- 相对路径优先 -->
<img src="./assets/logo.png" alt="logo" />

<!-- 带协议完整URL用于外部资源 -->
<a href="https://external.com">外部链接</a>
```

## 三、CSS 规范

### 8. 选择器

```css
/* ✅ ID 选择器少用，class 选择器为主 */
.page-container { }
.card-title { }

/* ❌ 避免嵌套过深 */
.nav .list .item .link .text { }

/* ✅ BEM 命名 */
.block__element--modifier { }
.modal__header--active { }
```

### 9. 单位使用

```css
/* 字体用 rem，便于无障碍 */
font-size: 1rem;

/* 间距用 rem 或 em */
padding: 1rem;

/* 媒体查询用 em（部分浏览器不支持 rem） */
@media (min-width: 48em) { }

/* 固定尺寸用 px */
border: 1px solid #000;

/* 动画用无单位数字（秒） */
transition: opacity 0.3s;
```

### 10. 简写属性

```css
/* ✅ */
margin: 0 auto;
padding: 10px 20px;
background: #fff url(bg.png) no-repeat center;
transition: all 0.3s;

/* 明确要覆盖的才拆开 */
margin-top: 0; /* 只需修改顶部时单独写 */
```

## 四、JavaScript / TypeScript 规范

### 11. 变量命名

```javascript
// ✅ 常量全大写下划线分隔
const MAX_RETRY_COUNT = 3;
const API_BASE_URL = 'https://api.example.com';

// ✅ 变量/函数 camelCase
let userName = '张三';
function getUserInfo() { }

// ✅ 类/构造函数 PascalCase
class UserService { }
function UserController() { }

// ✅ 布尔值用 is/has/can 前缀
let isActive = true;
let hasPermission = false;
```

### 12. 函数设计

```javascript
// ✅ 单一职责，函数名动宾短语
function getUserById(id) { }
function validateEmail(email) { }

// ✅ 避免副作用或明确说明
// 纯函数
function add(a, b) { return a + b; }
// 有副作用
function saveUserToDatabase(user) { /* 副作用 */ }

// ✅ 回调地狱避免，用 Promise/async-await
// ❌ 回调
api.getUser(id, (err, user) => { ... });

// ✅
const user = await api.getUser(id);
```

### 13. 类型约束（TypeScript）

```typescript
// ✅ 接口优于类型别名（可扩展）
interface User {
  id: number;
  name: string;
  email?: string; // 可选
}

// ✅ 用 enum 代替魔法字符串
enum UserRole {
  Admin = 'ADMIN',
  User = 'USER',
}

// ✅ 避免 any，用 unknown 代替需要检查的类型
function parseJSON(input: string): unknown {
  return JSON.parse(input);
}
```

### 14. 错误处理

```javascript
// ✅ 自定义错误类
class ValidationError extends Error {
  constructor(message, field) {
    super(message);
    this.name = 'ValidationError';
    this.field = field;
  }
}

// ✅ async 函数用 try-catch
async function fetchData() {
  try {
    const res = await api.getData();
    return res.data;
  } catch (err) {
    console.error('获取数据失败:', err);
    throw err;
  }
}
```

## 五、Java (Spring Boot) 规范

### 15. 包结构

```
com.example.project
├── controller      // 控制层
├── service         // 业务层
│   └── impl        // 实现
├── repository      // 数据访问层
├── model           // 实体类
│   ├── entity      // 数据库实体
│   ├── dto         // 数据传输对象
│   └── vo          // 视图对象
├── config          // 配置类
├── common          // 公共类
│   ├── exception   // 异常
│   └── constant    // 常量
└── util            // 工具类
```

### 16. RESTful API 命名

```

GET    /users/{id}     → 获取单个用户
POST   /users          → 创建用户

DELETE /users/{id}     → 删除用户
```

### 17. 配置管理

```yaml
# ❌ 硬编码

```

### 18. 事务处理

```java
// ✅ 在 Service 层加事务，粒度要细
@Service
public class UserService {
    @Transactional(rollbackFor = Exception.class)
    public void createUser(CreateUserDTO dto) {
        userRepository.save(dto);
        // 不要在事务中调用远程服务
    }
}
```

### 19. 日志规范


## 六、Python 规范

### 20. PEP 8 核心

```python
# ✅ 类名 PascalCase，函数/变量 snake_case
class UserService:
    def get_user_info(self, user_id):
        user_name = '张三'

# ✅ 常量大写
MAX_RETRY_COUNT = 3

import os

from datetime import datetime
```

### 21. 类型注解

```python
# ✅ 建议使用类型注解
def greet(name: str, times: int = 1) -> str:
    return (f"Hello, {name}! " * times)

# 类属性注解
class User:
    name: str
    age: int
    email: Optional[str] = None
```

### 22. 异常处理

```python
# ✅ 具体异常优先
try:
    result = division(a, b)
except ZeroDivisionError:
    result = 0

# ✅ 异常链保留原因
try:
    connect_database()
except ConnectionError as e:
    raise DatabaseError("无法连接数据库") from e

# ✅ 自定义异常
class ValidationError(Exception):
    pass
```

## 七、SQL 规范

### 23. 命名规范

```sql
-- 表名单数，字段全小写下划线
-- ✅
CREATE TABLE user (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_name VARCHAR(50) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    is_deleted TINYINT DEFAULT 0
);

-- ❌
CREATE TABLE UserInfo (
    UserID int,
    CreateTime datetime
);
```

### 24. 查询规范

```sql
-- ✅ 明确列名，避免 SELECT *
SELECT id, user_name, email FROM user;

-- ✅  WHERE 条件明确，加索引
SELECT * FROM user WHERE is_deleted = 0 AND status = 1;

-- ✅  LIMIT 限制结果集
SELECT * FROM logs LIMIT 100;

-- ❌ 大表全表扫描
SELECT * FROM million_rows_table;
```

### 25. 索引规范

```sql
-- ✅ 区分度高的列放前面
INDEX idx_status_created (status, created_at)

-- ✅ 复合索引遵循最左前缀原则
-- 查询 WHERE status = ? AND created_at > ? 可用
-- 查询 WHERE created_at > ? 不可用
```

## 九、安全规范（通用）

### 29. 敏感数据处理

```javascript
// ✅ 禁止在日志中打印敏感信息
// ❌
console.log('password:', password);
log.info('信用卡号:', cardNo);

// ✅ 脱敏处理
const maskCardNo = (cardNo) => cardNo.replace(/\d{13}/, '****');
```

### 30. SQL 注入防范

```java
// ❌ 字符串拼接 SQL
String sql = "SELECT * FROM user WHERE name = '" + name + "'";

// ✅ 参数化查询
@Query("SELECT u FROM User u WHERE u.name = :name")
User findByName(@Param("name") String name);
```

### 31. XSS 防范

```html
<!-- 模板引擎自动转义 -->
<!-- Vue: {{userInput}} 自动转义 -->
<!-- React: {userInput} 自动转义 -->

<!-- 确实需要 HTML 时手动处理 -->
function escapeHtml(str) {
    return str.replace(/[&<>"']/g, c => ({
        '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#39;'
    })[c]);
}
```
